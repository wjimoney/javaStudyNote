student s,
course c,
sc sc
WHERE s.sno=sc.sno
AND c.cno=sc.cno
AND s.sname='王五'

--显示内连接

student s 
JOIN sc sc ON s.sno=sc.sno
JOIN course c ON c.cno = sc.cno
AND s.sname='王五'



-- 5 where对原表数据做筛选  having是对select结果做筛选
SELECT s.sno,s.sname, AVG(sc.score)  from
student s,
course c,
sc sc
WHERE s.sno=sc.sno
AND c.cno=sc.cno
AND c.cstate=1
GROUP BY s.sno,s.sname
HAVING  AVG(sc.score)>90

-- 2 查询每班每科的平均分

SELECT t.class, m.sub, AVG(m.mark) FROM 
t_student t,
t_mark m
WHERE t.code=m.code
GROUP BY t.class, m.sub


-- 3 查询每班每科分数最高的同学  学科 分数
SELECT t.class, m.sub, max(m.mark) FROM 
t_student t,
t_mark m
WHERE t.code=m.code
GROUP BY t.class, m.sub


SELECT t.class, t.code,t.name, m.sub, m.mark FROM
t_student t,
t_mark m,
(
SELECT t.class, m.sub, max(m.mark) AS maxmark FROM 
t_student t,
t_mark m
WHERE t.code=m.code
GROUP BY t.class, m.sub
) tmp

WHERE t.code=m.code
AND tmp.class=t.class
AND tmp.sub=m.sub
AND tmp.maxmark=m.mark
